\section{Question No. 2.5}
A couple has two children,
\begin{enumerate}[itemsep=2pt]
\item Find the probability that both children are male, if it is known that at least one of the children is male.

\item Find the probability that both children are female, if it is known that at least one of the children is female.
\end{enumerate}
%
\section{Solution}
\begin{lemma}
If A and B are two events in a sample space S, then the Conditional Probability of A given B is defined as 
$P(A\mid B)$ =${\frac{P(A \cap B) }{P(B)}}$
\end{lemma}
From the given information we have,
\begin{table}[ht!]
\begin{tabular}{|l|l|}
\hline
Events & Description                         \\ \hline
$E1$      & Both children are male  \\ \hline
$F1$      & At least one child is male           \\ \hline
$E2$      & Both children are female  \\ \hline
$F2$      & At least one child is female            \\ \hline
$(E1\cap F1)$ & Both children are male \\
 & and at least one child is male 
\\ \hline
$(E1\cap F2)$ & Both children are female
and \\ & at least one child is female \\ \hline
\end{tabular}
\end{table}

Let boy be denoted by B and girl be denoted by G respectively.
\begin{align}
    S= \{ &BB,BG,GB,GG \}
\end{align}
\begin{enumerate}
\item We need to find P =$(E1\mid F1)$:
\begin{align}
 P(E1)=\frac{1}{4},P(F1)=\frac{3}{4} 
 \\
 P(E1\cap F1)=\frac{1}{4}
 \\
\therefore P(E1|F1)=\frac{1}{3}
\end{align}

\item We need to find P =$(E2\mid F2)$: 
\begin{align}
 P(E2)=\frac{1}{4},P(F2)=\frac{1}{2} 
 \\
 P(E2\cap F2)=\frac{1}{4}
 \\
\therefore P(E2|F2)=\frac{1}{2}
\end{align}
\end{enumerate}
